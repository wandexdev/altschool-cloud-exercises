# for setting up ansible remote connection
# Setup your ec2 with your ec2 key and download the key

# Setup /etc/ansible/hosts with the necessary info e.g
````
[webserver]
18.206.199.143 ansible_connection=ssh ansible_user=ubuntu ansible_ssh_private_key_file=/home/ubuntu/key.pem
````

# Update permission of ssh-key based on AWS Recommendation
- chmod 400 /home/ubuntu/key.pem

# ping hosts setup with ansible
- ansible all -m ping -v

# Run playbook
- ansible-playbook -i /etc/ansible/hosts /home/ubuntu/playbook.yml
